# Platform notes
# ──────────────
#  keyboard  → Windows / Linux only.  Do NOT install on macOS.
#  pynput    → macOS (and Linux).     Safe to install on Windows but not used.

#  afplay    → macOS built-in CLI tool; no install needed.

# permission: System Settings → Privacy & Security → Accessibility
pynput>=1.7.6
